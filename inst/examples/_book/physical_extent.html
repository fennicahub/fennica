library(dplyr)
library(knitr)

df_pagecount <- df.tmp %>%
  mutate(
    pagecount = suppressWarnings(as.numeric(pagecount)),
    volcount  = suppressWarnings(as.numeric(volcount)),
    pagecount_original_missing = is.na(pagecount.page.info) | pagecount.page.info == "",
    pagecount_final_missing = is.na(pagecount),
    volcount_missing = is.na(volcount)
  )

n_total <- nrow(df_pagecount)

pagecount_summary <- df_pagecount %>%
  summarise(
    documents = n(),
    pagecount_available = sum(!is.na(pagecount)),
    pagecount_missing = sum(is.na(pagecount)),
    pagecount_available_percent = round(100 * pagecount_available / documents, 1),
    mean_pagecount = round(mean(pagecount, na.rm = TRUE), 1),
    median_pagecount = median(pagecount, na.rm = TRUE),
    min_pagecount = min(pagecount, na.rm = TRUE),
    max_pagecount = max(pagecount, na.rm = TRUE),
    volcount_available = sum(!is.na(volcount)),
    volcount_missing = sum(is.na(volcount)),
    mean_volcount = round(mean(volcount, na.rm = TRUE), 1),
    median_volcount = median(volcount, na.rm = TRUE)
  )

df_pagecount %>%
  mutate(
    volume_type = case_when(
      is.na(volcount) ~ "Missing volume count",

  group_by(volume_type) %>%

These estimates are used to fill in page count info for the remaining documents where page count info is missing.

The multi-volume documents average page counts are given per volume.

The page count estimates are calculated without plates. Plate information is added separately for each document on top of the page count estimate.
